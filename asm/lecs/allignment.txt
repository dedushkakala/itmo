На стеке:
	в 64 битном - стек поинтер должен быть выравнен на значение кратном 16, для того, чтобы писать хмм команды по стеку

выделить n байт на стеке с выравниванием по 16
push ebp
mov ebp, esp
sub esp, n
and esp, -16
moc esp, ebp
pop ebp
